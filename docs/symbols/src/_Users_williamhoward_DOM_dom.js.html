1 /**
  2   
  3   Tiny DOM templating for javascript. Syntax stolen from MochiKit, because I like the way it looks.
  4 
  5 	Looks something like:
  6 
  7 	DIV({"class":"somClass"},
  8 		A({"href":"http://google.com"}, my link), SPAN(null,"some text")
  9 	);
 10  
 11   @namespace
 12  */
 13 var DOM = (function() {
 14 
 15 	/**
 16 	  Creates a new DOM object.
 17 	  @function
 18 	  @param {String} tag Name of the tag (like, "A", "DIV", "IMG").
 19 	  @param {Object} attrs Attributes for element (in object notation, like: {id:"someId", class:"someClass"})
 20 	  @param {String} txt Text from either previous createDOM's, .
 21 		@returns {String} The text for the HTML element.
 22 	 */
 23 	var createDOM = function(tag, attrs, txt) {
 24 
 25 		var args = Array.prototype.slice.call(arguments, 0);
 26 
 27 		if (args.length > 0) {
 28 			if (typeof(txt) === "undefined" || !txt) { //This is a singleton element.
 29 				return '<'+tag,attrsToStr(attrs)+' />';
 30 			} else {
 31 				return '<'+tag+attrsToStr(attrs)+'>'+args.slice(2).join("")+'</'+tag+'>';
 32 			}
 33 		}
 34 
 35 	};
 36 
 37 	/**
 38 	  Turns attributes object to a string.
 39 	  @function
 40 	  @param {Object} attrs Attributes to turn into an object.
 41 		@returns {String} The attributes as a string.
 42 	 */
 43 	var attrsToStr = function(attrs) {
 44 		var str = '';
 45 		for (a in attrs) {
 46 			str += ' '+ a +'="'+ attrs[a]+ '"';
 47 		}
 48 		return str;
 49 	};
 50 
 51 	/**
 52 	  Creates a shortcut for creating DOM elements, like A({"href":"http://google.com"}, "link") instead of DOM.createDOM("a",{"href":"http://google.com"}, "link").
 53 	  @function
 54 	  @param {String} tag Name of the tag to create a shortcut for.
 55 	 */
 56 	var DOMShortcut = function(tag) {
 57 		return function() {
 58 			return createDOM.apply(this, [tag].concat(Array.prototype.slice.call(arguments, 0)));
 59 		};
 60 	};
 61 
 62 	return {
 63 		createDOM : createDOM,
 64 		DOMShortcut : DOMShortcut
 65 	}
 66 
 67 })();
 68 
 69 var A      = DOM.DOMShortcut("A"),
 70     DIV    = DOM.DOMShortcut("DIV"),
 71     SPAN   = DOM.DOMShortcut("SPAN"),
 72     INPUT  = DOM.DOMShortcut("INPUT"),
 73     TR     = DOM.DOMShortcut("TR"),
 74     TD     = DOM.DOMShortcut("TD"),
 75     TABLE  = DOM.DOMShortcut("TABLE"),
 76     THEAD  = DOM.DOMShortcut("THEAD"),
 77     TBODY  = DOM.DOMShortcut("TBODY"),
 78     UL     = DOM.DOMShortcut("UL"),
 79     LI     = DOM.DOMShortcut("LI"),
 80     OL     = DOM.DOMShortcut("OL"),
 81     P      = DOM.DOMShortcut("P"),
 82     STRONG = DOM.DOMShortcut("STRONG"),
 83     EM     = DOM.DOMShortcut("EM");
